23. Merge K-Sorted Lists

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */

// Rotating head of the list into heap as we popping off heap
class Solution{
    public ListNode mergeKLists(ListNode[] lists){
        if(lists.length == 0) return null;
        PriorityQueue<ListNode> heap = new PriorityQueue<>(lists.length, (a,b) -> Integer.compare(a.val, b.val));
        
        for(ListNode i : lists) if(i != null)heap.offer(i);
        
        ListNode res = new ListNode(0), temp = res;
        while(!heap.isEmpty()){
            temp.next = heap.poll();
            temp = temp.next;
            if(temp.next != null) heap.offer(temp.next);
        }
        return res.next;
        
        
    }
}
