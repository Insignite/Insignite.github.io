329. Longest Increasing Path

// https://leetcode.com/problems/longest-increasing-path-in-a-matrix/

class Solution {
    public int longestIncreasingPath(int[][] matrix) {
        int row = matrix.length;
        if(row == 0) return 0;
        int col = matrix[0].length;
        if(col == 0) return 0;
        
        int max = 0;
        int[][] mem = new int[row][col];
        for(int i = 0; i < matrix.length; i++){
            for(int j = 0; j < matrix[0].length; j++){
                max = Math.max(max, helper(matrix, i, j, mem, -1));
            }
        }
        return max;
    }
    
    private int helper(int[][] matrix, int row, int col, int[][] mem, int lastValue){
        if(row < 0 || row >= matrix.length || 
           col < 0 || col >= matrix[0].length || 
           matrix[row][col] <= lastValue) return 0;
        // if already calculate, return;
        if(mem[row][col] != 0) return mem[row][col];
        
        int d = helper(matrix, row + 1, col, mem, matrix[row][col]);
        int u =  helper(matrix, row - 1, col, mem, matrix[row][col]);
        int r = helper(matrix, row, col + 1, mem, matrix[row][col]);
        int l = helper(matrix, row, col - 1, mem, matrix[row][col]);
        mem[row][col] = Math.max(d, Math.max(u, Math.max(r, l)))+ 1;
        return mem[row][col];
        
        
    }
}
