42. Trapping Rain Water

// https://leetcode.com/problems/trapping-rain-water/

class Solution {
    public int trap(int[] height){
        int sum = 0;
        int lo = 0, hi = height.length - 1;
        
        int left = 0, right = 0;
        // keep track of max height left and right water height
        // then subtract all the height lower than that current max boundary height
        // if we meet a height higher than current max, new height get set
        // cuz we have fill the water into the terrain of last max height and new max height
        while(lo < hi){
            left = Math.max(left, height[lo]);
            right = Math.max(right, height[hi]);
            
            if(left < right){
                sum += left - height[lo];
                lo++;
            } else{
                sum += right - height[hi];
                hi--;
            }
        }
        
        return sum;
    }
}
