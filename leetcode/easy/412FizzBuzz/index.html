412. FizzBuzz

// https://leetcode.com/problems/fizz-buzz/submissions/

class Solution {
    public List<String> fizzBuzz(int n) {
        List<String> res = new LinkedList<>();
        for(int i = 1; i <= n; i++){
            if(i%3 == 0){
                res.add("Fizz");
                if(i%5 == 0) res.set(i-1, "FizzBuzz");
            }
            else if(i%5 == 0) res.add("Buzz");
            else res.add(Integer.toString(i));
            
        }
        
        return res;
    }
}
