929. Unique Email Addresses

// https://leetcode.com/problems/unique-email-addresses/

class Solution {
    public int numUniqueEmails(String[] emails) {
        HashSet<String> res = new HashSet<>();
        for(String i : emails){
            StringBuilder build = new StringBuilder();
            int pt1 = 0;
            for(int j = 0; j < i.length(); j++){
                char temp = i.charAt(j);
                if(temp == '.') continue;
                if(temp == '@' || temp == '+'){
                    build.append(i.substring(i.indexOf('@'),i.length()));
                    break;
                }
                // if(temp == '+'){
                //     pt1 = j;
                //     while(i.charAt(pt1) != '@' && pt1 < i.length()) pt1++;
                //     build.append(i.substring(pt1, i.length()));
                //     break;
                // }
                // if(temp == '@'){
                //     build.append(i.substring(j, i.length()));
                //     break;
                // }
                
                build.append(temp);
            }
            res.add(build.toString());
        }
        
        return res.size();
    }
}
