326. Power of Three

// https://leetcode.com/problems/power-of-three/

class Solution {
    public boolean isPowerOfThree(int n) {
        // n = 3^i
        // 3^i = 3^(log_3 MaxInt)
        int max = (int) Math.pow(3, (int)(Math.log(Integer.MAX_VALUE)/Math.log(3)));
        return (n > 0 && max%n== 0);
    }
}
