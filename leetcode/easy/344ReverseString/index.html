344. Reverse String

// https://leetcode.com/problems/reverse-string/

class Solution {
    public void reverseString(char[] s) {
        int ptr = s.length-1;
        for(int i = 0; i <= ptr; i++){
            if(ptr < i) break;
            char temp = s[i];
            s[i] =  s[ptr];
            s[ptr] = temp;
            ptr--;
        }
    }
}
