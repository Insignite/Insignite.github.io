788. Rotated Digits

// https://leetcode.com/problems/rotated-digits/

class Solution {
    public int rotatedDigits(int N) {
        int[] rotate = new int[]{0,0,1,-1,-1,1,1,-1,0,1};
        int counter = 0;
        for(int i = 1; i <= N; i++){
            int compare = i;
            boolean different = false;
            while(compare > 0 ){
                int temp = compare % 10;
                if(rotate[temp] == -1){
                    different = false;
                    break;
                }
                else if(rotate[temp] == 1) different = true;
                compare = compare/10;
            }
            if(different) counter++;
        }
        return counter;
    }
}
