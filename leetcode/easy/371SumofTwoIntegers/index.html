371. Sum of Two Integers

// https://leetcode.com/problems/sum-of-two-integers/

class Solution {
    public int getSum(int a, int b) {
        int res = 0;
        while(b != 0){
            res = a&b;
            a = a ^ b;
            b = res<<1;
        }
        return a;
    }
}
