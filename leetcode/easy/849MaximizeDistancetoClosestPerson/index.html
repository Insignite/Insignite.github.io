849. Maximize Distance to Closest Person

// https://leetcode.com/problems/maximize-distance-to-closest-person/

class Solution {
    public int maxDistToClosest(int[] seats) {
        int result = 0, size = seats.length, win = -1;
        
        for(int i = 0; i < size; i++){
            if(seats[i] == 1){
                int distance = (i - win)/2;
                // if haft distance from 1 to another 1 is less than the result return
                result = win < 0 ? i : result < distance ? distance : result;
                // Update the window start point
                win = i;
            }
        }
        // this check the assumption we don't encounter 1 at the end;
        result = result > size - win - 1 ? result : size - win - 1;
        return result ;
    }
}
