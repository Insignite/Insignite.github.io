13. Roman To Integer

// https://leetcode.com/problems/roman-to-integer/
class Solution {
    public int romanToInt(String s){
        // discussion give idea about using
        // array instead of hashmap
        // it's much faster
        
        int[] map = new int[26];
        map['I' - 'A'] = 1;
        map['V' - 'A'] = 5;
        map['X' - 'A'] = 10;
        map['L' - 'A'] = 50;
        map['C' - 'A'] = 100;
        map['D' - 'A'] = 500;
        map['M' - 'A'] = 1000;
        
        char pre = 'A';
        int sum = 0;
        
        
        for(char i : s.toCharArray()){
            // the idea is that if the current value is less than previous
            // it mean that the value could fall under IV, IX, XL, etc.
            // so we subtract double it out from current sum
            if(map[i - 'A'] > map[pre - 'A']) sum -= 2 * map[pre - 'A'];
            
            sum += map[i - 'A'];
            pre = i;
        }
        return sum;
    }
