204. Count Primes

// https://leetcode.com/problems/count-primes/

class Solution {
    public int countPrimes(int n) {
        if(n <= 2) return 0;
        boolean[] res = new boolean[n+1];
        res[2] = true;
        for(int i = 3; i < n; i += 2) res[i] = true;
        
        for(int j = 3; j*j <= n; j++){
            if(res[j]){
                for(int k = j + j; k<=n; k+= j){
                    res[k] = false;
                }
            }
        }
        int counter = 1;
        for(int m = 3; m < n; m += 2) counter += res[m] ? 1 : 0;
        return counter;
    }
}
