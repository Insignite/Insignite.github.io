268. Missing Number

// https://leetcode.com/problems/missing-number/

class Solution {
    public int missingNumber(int[] nums) {
        int total = 0, largest = 0;
        
        for(int i = 0; i < nums.length; i++){
            total += nums[i];
            largest += i;
        }
        largest += nums.length;
        
        return largest - total;
    }
}
