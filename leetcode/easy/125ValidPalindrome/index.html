125. Valid Palindrome

// https://leetcode.com/problems/valid-palindrome/


// Try best not to use regex or any other built in function for this problem
class Solution {
    public boolean isPalindrome(String s) {
        int index = s.length() -1;
        s = s.toLowerCase();
        for(int i = 0; i < s.length() && index >= 0; i++){
            char temp1 = s.charAt(i);
            char temp2 = s.charAt(index);
            if(temp1 < '0' || temp1 > 'z' || (temp1 > '9' && temp1 < 'a')) {
                continue;
            }
            while(temp2 < '0' || temp2 > 'z' || (temp2 > '9' && temp2 < 'a')){
                index--;
                temp2 = s.charAt(index);
            }
            
            if(temp1 != temp2){
                return false;
            }
            index--;
        }
        
        return true;

    }
}
