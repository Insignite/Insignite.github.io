1295. Find Numbers with Even Number of Digits

// https://leetcode.com/problems/find-numbers-with-even-number-of-digits/

class Solution {
    public int findNumbers(int[] nums){
        int cnt = 0, digit = 1;
        for(int i : nums){
            while(i >= 10){
                i /= 10;
                digit++;
            }
            if(digit %2 == 0) cnt++;
            digit = 1;
        }
        return cnt;
    }
