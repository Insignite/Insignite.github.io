169. Majority Element

// https://leetcode.com/problems/majority-element/submissions/


// HashMap Solution
class Solution {
    public int majorityElement(int[] nums) {
        int size = nums.length, largest = 0;
        HashMap<Integer, Integer> holder = new HashMap<>();
        for(int i : nums){
            if(holder.containsKey(i)) holder.put(i, holder.get(i) + 1);
            else holder.put(i, 1);
            if(holder.get(i) > size/2){
                largest = i;
                break;
            }
        }
        return largest;
    }
}

// Boyer-Moore Algorithm
class Solution {
    public int majorityElement(int[] nums) {
        int counter = 0, result = 0;
        
        for(int i : nums){
            if(counter == 0) result = i;
            if(result != i) counter--;
            else counter++;
        }
        
        return result;
    }
}
