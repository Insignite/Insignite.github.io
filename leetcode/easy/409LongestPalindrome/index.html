409. Longest Palindrome

//  https://leetcode.com/problems/longest-palindrome/

class Solution {
    public int longestPalindrome(String s) {
        int[] charS = new int[128];
		int count = 0;
		
		// Count how repetition of letters in a given string
		for(int i = 0; i< s.length(); i++) {
			charS[s.charAt(i)]++;
		}
		for(int num : charS) {
			if(count %2 != 0){
                count += (num / 2) * 2;
            }
            else{
                count += num;
            }
		}
        
		return count;
    }
}
