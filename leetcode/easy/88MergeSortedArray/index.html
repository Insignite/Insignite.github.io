88. Merge Sorted Array

// https://leetcode.com/problems/merge-sorted-array/

class Solution {
    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int p1 = m - 1, p2 = n - 1, size = m + n -1;
        while(p1 >= 0 && p2 >= 0){
            if(nums1[p1] > nums2[p2]) nums1[size--] = nums1[p1--];
            else nums1[size--] = nums2[p2--];
        }
        
        while(p2 >= 0) nums1[size--] = nums2[p2--];
