394. Decode String

// https://leetcode.com/problems/decode-string/

class Solution {
    public String decodeString(String s){
        int size = s.length(), num = 0;
        char c = ' ';
        String res = "", holder = "";
        Stack<Integer> stackNum = new Stack<>();
        Stack<String> stackStr = new Stack<>();
        for(int i = 0; i < size; i++){
            c = s.charAt(i);
            if(c >= '0' && c <= '9'){
                num = 0;
                while(c >= '0' && c <= '9'){
                    num = num * 10 + (c - '0');
                    i++;
                    c = s.charAt(i);
                }
                stackNum.push(num);   
                i--;
            }
            else if(c == '['){
                stackStr.push(res);
                res = "";
                
            }
            else if(c == ']'){
                StringBuilder temp = new StringBuilder(stackStr.pop());
                num = stackNum.pop();
                for(int j = 0; j < num; j++) temp.append(res);
                res = temp.toString();
            }
            else res += c;
        }
        return res;
    }
}
