49. Group Anagrams

// https://leetcode.com/problems/group-anagrams/

class Solution {
    public List<List<String>> groupAnagrams(String[] strs) {
        HashMap<String, List<String>> table = new HashMap<>();
        List<List<String>> res = new ArrayList<>();
        
        for(String i : strs){
            int[] hash = new int[26];
            for(char j : i.toCharArray()){
                hash[j - 'a']++;
            }
            String temp = Arrays.toString(hash);
            if(!table.containsKey(temp)) table.put(temp, new ArrayList());
            table.get(temp).add(i);
        }

        res.addAll(table.values());
        return res;

    }
}
