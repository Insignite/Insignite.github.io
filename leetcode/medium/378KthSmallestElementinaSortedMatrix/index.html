378. Kth Smallest Element In a Sorted Matrix

// https://leetcode.com/problems/kth-smallest-element-in-a-sorted-matrix/

class Solution {
    
    // The user of binary search
    // Clearer explaination from arslanah can be find by follow the link
    // https://leetcode.com/problems/kth-smallest-element-in-a-sorted-matrix/discuss/301357/Java-0ms-(added-Python-and-C%2B%2B)%3A-Easy-to-understand-solutions-using-Heap-and-Binary-Search
    public static int kthSmallest(int[][] matrix, int k){
        int size = matrix.length;
        int lo = matrix[0][0], hi = matrix[size-1][size-1];
      
        while(lo <= hi){
            int mid = lo + (hi - lo)/2;
            int count = helper(matrix, mid);
            if(count < k) lo = mid + 1;
            else hi = mid - 1;
        }
        
        return lo;
    
    }
    
    private static int helper(int[][] matrix, int mid){
        int count = 0;
        int size = matrix.length, lo = 0, hi = matrix.length - 1;
        while(lo < size && hi >= 0){
            if(matrix[lo][hi] > mid) hi--;
            else{
                count += hi + 1;
                lo++;
            }
        }
        return count;
    }
    
    
}
