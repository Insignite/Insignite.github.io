134. Gas Stations

// https://leetcode.com/problems/gas-station/

class Solution {
    public int canCompleteCircuit(int[] gas, int[] cost) {
        int gasSize = gas.length;
        if(gasSize == 1 && gas[0] < cost[0]) return -1;

        int previous = 0, index = 0, current = 0;
        
        for(int i = gasSize - 1; i >= 0; i--){
            current += gas[i] - cost[i];
            if(current > previous){
                previous = current;
                index = i;
            }
        }
        
        return current < 0 ? -1 : index;
    
    }
}
