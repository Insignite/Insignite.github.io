36. Valid Sudoku

// https://leetcode.com/problems/valid-sudoku/

class Solution {
    public boolean isValidSudoku(char[][] board) {
        int size = 9;
        boolean[][] ver = new boolean[size][size];
        boolean[][] hor = new boolean[size][size];
        boolean[][] block = new boolean[size][size];
        for(int i = 0; i < size; i++){
            for(int j = 0; j < size; j++){
                if(board[i][j] == '.') continue;
                
                int temp = board[i][j] - '1';
                
                if(ver[i][temp] || hor[j][temp] || block[(i/3)*3 + (j/3)][temp]) return false;
                ver[i][temp] = hor[j][temp] = block[(i/3)*3 + (j/3)][temp] = true;
            }
        }
        return true;
    }
}
