152. Maximum Product Subarray

// https://leetcode.com/problems/maximum-product-subarray/

class Solution {
    public int maxProduct(int[] nums) {
        int size = nums.length;
        if(size == 0) return 0;
        if(size == 1) return nums[0];
        
        int max = 0, lo = 1, hi = 1;
        for(int i = 0, j = size - 1; i < size; i++, j--){
            lo *= nums[i];
            hi *= nums[j];
            max = max > lo ? max : lo;
            max = max > hi ? max : hi;
            if(lo == 0) lo = 1;
            if(hi == 0) hi = 1;
        }
        
        return max;
    }
}
