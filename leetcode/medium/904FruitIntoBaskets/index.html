904. Fruit Into Baskets

// https://leetcode.com/problems/fruit-into-baskets/

class Solution {
    public int totalFruit(int[] tree) {

        // Using sliding windows method
        // poorly word problem, this can view as
        // what is longest sub-set you can make 
        // with N-elements continously, in this cas
        // it is 2 cuz basket only allow 2 fruits.
        if(tree == null || tree.length == 0) return 0;
        Map<Integer, Integer> basket = new HashMap<>();
        int lo = 0, max = 0;
        for(int hi = 0; hi < tree.length; hi++){
            basket.put(tree[hi], basket.getOrDefault(tree[hi], 0) + 1);
            
            while(basket.size() > 2){
                basket.put(tree[lo], basket.get(tree[lo])- 1);
                basket.remove(tree[lo++], 0);
            }
            max = Math.max(max, hi - lo +1);
        }
        return max;
    }
}
