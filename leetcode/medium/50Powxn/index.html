50. Pow(x,n)


// https://leetcode.com/problems/powx-n/

class Solution {
    public double myPow(double x, int n) {
        double res = 1.0;
        int pow = n;
        if(pow < 0){
            pow = -pow;
            x = 1/x;
        }
        
        while(pow != 0){
            if((pow & 1) == 1){
                res *= x;
            }
            x *= x;
            pow >>>=1;
        }
        
        return res;
    }
}
