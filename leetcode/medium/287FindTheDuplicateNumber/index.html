287. Find The Duplicate Number

// https://leetcode.com/problems/find-the-duplicate-number/

class Solution {
    public int findDuplicate(int[] nums) {
        int pt1 = 0, pt2 = 0;
        
        do{
            pt1 = nums[pt1];
            pt2 = nums[nums[pt2]];
        }while(pt1 != pt2);
        
        pt1 = 0;
        while(pt1 != pt2){
            pt1 = nums[pt1];
            pt2 = nums[pt2];
        }
        
        return pt1;
    }
}
