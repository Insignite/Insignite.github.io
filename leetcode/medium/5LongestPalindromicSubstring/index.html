5. Longest Palindromic Substring

// https://leetcode.com/problems/longest-palindromic-substring/

class Solution {
    int max = 0;
    String res = "";
    public String longestPalindrome(String s) {
        int size = s.length();
        if(size == 0) return "";
        
        char[] charArr = s.toCharArray();
        for(int i = 0; i < size; i++){
            helper(charArr, size, i, 0);
            helper(charArr, size, i, 1);
        
        }
        
        return res;
    }
    
    private void helper(char[] charArr, int length, int index, int shifter){
        int lo = index, hi = index + shifter;
        
        while(lo >= 0 && hi < length && charArr[lo] == charArr[hi]){
            lo--;
            hi++;
        }
        
        if(hi - lo - 1 > max){
            max =  hi - lo - 1;
            res = new String(charArr, lo + 1, max);
        }
    }
}
