347. Top K Frequent Element

// https://leetcode.com/problems/top-k-frequent-elements/

class Solution {
    public List<Integer> topKFrequent(int[] nums, int k) {
        int size = nums.length;
        HashMap<Integer, Integer> temp = new HashMap<>();
        for(int i : nums){
            // if(temp.containsKey(i)) temp.put(i, temp.get(i) + 1);
            // else temp.put(i, 1);
            
            // Cleaner version
            temp.put(i, temp.getOrDefault(i, 0) + 1);

        }
        List<Integer>[] adjMatrix = new List[size + 1];
        
        for(Map.Entry<Integer, Integer> value : temp.entrySet()){
            int pt = value.getValue();
            if(adjMatrix[pt] == null) adjMatrix[pt] = new LinkedList<>();
            adjMatrix[pt].add(value.getKey());
        }
        
        
        List<Integer> result = new LinkedList<>();
        for(int i = adjMatrix.length -1; i > 0 && k > 0; --i){
            if(adjMatrix[i] != null){
                for(int j = 0; j < adjMatrix[i].size(); j++){
                    result.add(adjMatrix[i].get(j));

                }
                k -= adjMatrix[i].size();
            }
        }
        return result;
    }
}
