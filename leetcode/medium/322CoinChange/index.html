322. Coin Change

// https://leetcode.com/problems/coin-change/

import java.util.Arrays;

class Solution {
    int min_count = Integer.MAX_VALUE;
    public int coinChange(int[] coins, int amount) {
        // Edge case
        if(amount == 0) return 0;
        if(coins.length == 0) return -1;
        Arrays.sort(coins);
        helper(coins, amount, coins.length -1, 0);
        return min_count == Integer.MAX_VALUE ? -1 : min_count;
    }
    
    
    void helper(int[] coins, int target, int index, int count){
        // Return case of recursive
        if(index < 0 || count >= min_count - 1) return;
        // Loop to get the biggest numbers of time a coin can be use
        for(int i = target/coins[index]; i >= 0; i--){
            int remainder = target - (i * coins[index]);
            int temp_count = count + i;
            
            // As long as not reach target value and current coin not bigger 
            //than the smallest amount of coin can be use, 
            //then recursive calling, else break out because it doesn't
            // matter when the current count bigger than the min_count
            if(remainder > 0 && temp_count < min_count) helper(coins, remainder, index -1, temp_count);
            else{
                min_count = (remainder == 0 && temp_count < min_count) ? temp_count : min_count;
                break;
            }
        }
    }
}
