75. Sort Colors

// https://leetcode.com/problems/sort-colors/

class Solution {
    public void sortColors(int[] nums){
        int lo = 0, mi = 0, hi = 0;
        
        for(int i : nums){
            if(i == 0) lo++;
            if(i == 1) mi++;
            if(i == 2) hi++;
        }
        
        for(int i = 0; i < lo; i++) nums[i] = 0;
        for(int i = 0; i < mi; i++) nums[i + lo] = 1;
        for(int i = 0; i < hi; i++) nums[i + lo + mi] = 2;
        
    }
}
