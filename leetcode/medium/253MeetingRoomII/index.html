253. Meeting Room II

// https://leetcode.com/problems/meeting-rooms-ii/

class Solution {
    public int minMeetingRooms(int[][] intervals){
        int size = intervals.length;
        if(size == 0) return 0;
        if(size == 1) return 1;
        int[] start = new int[size], end = new int[size];
        
        int count = 0, res = 0;
        for(int[] i : intervals){
            start[count] = i[0];
            end[count] = i[1];
            count++;
        }
        
        Arrays.sort(start);
        Arrays.sort(end);
        count = 0;
        for(int i = 0; i < size; i++){
            if(start[i] < end[count]) res++;
            else count++;
        }
        return res;
    }
