40. Combination Sum II

// https://leetcode.com/problems/combination-sum-ii/

class Solution {
    public List<List<Integer>> combinationSum2(int[] candidates, int target) {
        List<List<Integer>> res = new ArrayList<>();
        
        if(candidates == null || candidates.length == 0) return res;
        
        Arrays.sort(candidates);
        List<Integer> combine = new ArrayList<>();
        Combination(candidates, res, combine, target, 0);
        
        return res;
    }
    
    private static void Combination(int[] candidates, List<List<Integer>> result, List<Integer> combine, int target, int index){
        if(target == 0){
            result.add(new ArrayList<>(combine));
            return;
        }
        
        for(int i = index; i < candidates.length; i++){
            if(i != index && candidates[i] == candidates[i - 1]) continue;
            
            if(candidates[i] > target) break;
            
            combine.add(candidates[i]);
            Combination(candidates, result, combine, target - candidates[i], i + 1);
            combine.remove(combine.size() - 1);
        }
    }
}
