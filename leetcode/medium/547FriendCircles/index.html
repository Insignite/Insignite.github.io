547. Friend Circles

// https://leetcode.com/problems/friend-circles/

import java.util.*;
class Solution {
    public int findCircleNum(int[][] M) {
        boolean[] visited = new boolean[M.length];
        int count = 0;
        for(int i = 0; i < M.length; i++){
            if(!visited[i]){
                dfsing(M, visited, i);
                count++;
            }
        }
        return count;
    }
    
    public void dfsing(int[][] m, boolean[] visited, int i){
        for(int j = 0; j < m.length; j++){
            if(m[i][j] == 1 && !visited[j]){
                visited[j] = true;
                dfsing(m, visited, j);
            }
        }
    }

}
