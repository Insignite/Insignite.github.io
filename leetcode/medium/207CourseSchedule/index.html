207. Course Schedule

// https://leetcode.com/problems/course-schedule/

class Solution {
    public boolean canFinish(int numCourses, int[][] prere){
        // Creating adjacency list;
        ArrayList<Integer>[] course = new ArrayList[numCourses];
        int counter = 0;
        int[] classWithoutPre = new int[numCourses];
        ArrayList<Integer> classToGoThru = new ArrayList<>();
        for(int i = 0; i < numCourses; i++){
            course[i] = new ArrayList<Integer>();
        }
        
        // Adding node into the adj list
        for(int[] temp: prere){
            course[temp[1]].add(temp[0]);
            classWithoutPre[temp[0]]++;
        }
        
        //Getting class without prereq
        for(int i =0; i < classWithoutPre.length; i++){
            if(classWithoutPre[i] == 0){
                classToGoThru.add(i);
                counter++;
            }
        }
        for(int i = 0; i < classToGoThru.size(); i++){
            for(int j: course[classToGoThru.get(i)]){
                classWithoutPre[j]--;
                if(classWithoutPre[j] == 0){
                    classToGoThru.add(j);
                    counter++;
                }
            }
        }
        return counter == numCourses;
        
        
    }
}
